from browser import html, document

# 定義 URL，指向學員資料的網頁
url = "https://mde.tw/list/2b.txt"

# 使用 open() 方法從 URL 讀取數據並轉換為數列
data = open(url).readlines()

# 濾除標題資料，從第二行開始取數
all_stud = data[1:]

# 利用 document[] 取得 id 為 "brython_div1" 的位置
brython_div1 = document["brython_div1"]

# 顯示學員總數
brython_div1 <= "總共有 " + str(len(all_stud)) + " 名學員"
brython_div1 <= html.BR()

# 使用迴圈遍歷每個學員的資料
for i, student in enumerate(all_stud):
    # 假設每行資料的格式是 "學號\tGitHub倉儲名稱"
    student_data = student.strip().split('\t')

    if len(student_data) == 2:  # 確保資料格式正確
        student_id = student_data[0]
        github_name = student_data[1]

        # 構造 URL
        personal_site = f"https://mdecd2025.github.io/hw-{github_name}/content/index.html"
        github_repo = f"https://github.com/mdecd2025/hw-{github_name}"

        # 創建超連結
        personal_link = html.A(personal_site, href=personal_site, target="_blank")  # target="_blank" 讓鏈接在新頁籤打開
        github_link = html.A(github_repo, href=github_repo, target="_blank")

        # 顯示學員資料，依照要求的格式，並將 URL 顯示為超連結
        brython_div1 <= f'"{student_id}" '
        brython_div1 <= personal_link  # 插入個人網站超連結
        brython_div1 <= " "  # 空格
        brython_div1 <= github_link  # 插入 GitHub 倉儲超連結
        brython_div1 <= html.BR()
    else:
        # 當資料格式不正確時，顯示錯誤訊息
        brython_div1 <= f"資料錯誤: {student}"
        brython_div1 <= html.BR()

籃框架被配置在一定範圍內, 可隨機慢速前進、後退及左右擺動, 投籃機構系統帶有一定數量的籃球, 被配置在可自由移動的輪車上.

操作者可利用鍵盤特定按鍵控制投籃輪車的移動並發射投籃, 每投出一球後系統透過記分板進行計分, 並由送球機構進行補球或移動輪車取球, 遊戲可進行至全部數量籃球投完為止.
